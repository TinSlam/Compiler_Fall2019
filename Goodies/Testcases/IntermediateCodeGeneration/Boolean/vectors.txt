class Program{
	static void _main(){
		real sum;
		real something;
		real length1;
		real length2;
		
		sum = 0;
		
		length1 = 0;
		for(i in 0 to 8 steps 2){ // 0 2 4 6
			length1 = length1 + i * i;
			
			length2 = 0;
			for(j in 0 to 4 steps 1){ // 0 1 2 3
				length2 = length2 + j * j;

				sum = sum + (i * j);
			}
		}
			
		something = sum / (length1 * length2);
		
		print("{something}");
	}
}
